def free_food(data, var_kcal, var_prot):
    data_2013 = data.loc[data["year"]==2013]
    data_2017 = data.loc[data["year"]==2017]
    
    KCAL_2013 = float(data_2013.groupby("year").sum()[var_kcal]) / KCAL_CAPITAYEAR
    KCAL_2017 = float(data_2017.groupby("year").sum()[var_kcal]) / KCAL_CAPITAYEAR
    PROT_2013 = float(data_2013.groupby("year").sum()[var_prot]) / PRO_KGAYEAR
    PROT_2017 = float(data_2017.groupby("year").sum()[var_prot]) / PRO_KGAYEAR
    
    supply=pd.DataFrame({"population": [POPULATION_2013, POPULATION_2017],
                         "A": [KCAL_2013, KCAL_2017],
                         "B": [PROT_2013, PROT_2017]},
                        index=[2013, 2017])
    
    supply["C"] = round(100 * supply["A"] / supply["population"], 2)
    supply["D"] = round(100 * supply["B"] / supply["population"], 2)
    
    supply.columns=["population", 
                    "Nombre d'êtres humains pouvant être nourris en calories",
                    "Nombre d'êtres humains pouvant être nourris en protéines",
                    "Pourcentage de la population mondiale pouvant être nourris en calories", 
                    "Pourcentage de la population mondiale pouvant être nourris en protéines"]
    
    return(supply)
